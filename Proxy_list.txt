# --- KEEP ALIVE ---

def run_web():

# --- STORAGE ---
BASE = "user_files"
os.makedirs(BASE, exist_ok=True)
running = {}

# --- BOT COMMANDS ---
@bot.message_handler(commands=['start'])
def start(msg):
    if msg.from_user.id == OWNER_ID:
        bot.reply_to(msg, "✅ سیستەم ب سەرکەفتیانە ژ گیت هاب دهێتە رەن کرن!")

@bot.message_handler(content_types=['document'])
def upload(msg):
    if msg.from_user.id != OWNER_ID: return
    file_info = bot.get_file(msg.document.file_id)
    downloaded = bot.download_file(file_info.file_path)
    user_dir = f"{BASE}/{msg.from_user.id}"
    os.makedirs(user_dir, exist_ok=True)
    path = f"{user_dir}/{msg.document.file_name}"
    with open(path, "wb") as f: f.write(downloaded)
    bot.reply_to(msg, f"📥 فایل هاتە پاراستن: {msg.document.file_name}")

@bot.message_handler(func=lambda m: m.text and m.text.startswith("run"))
def run_file(msg):
    if msg.from_user.id != OWNER_ID: return
    try:
        file_name = msg.text.split(" ")[1]
        path = f"{BASE}/{msg.from_user.id}/{file_name}"
        if os.path.exists(path):
            def runner():
                while path in running:
                    p = subprocess.Popen(["python", path])
                    running[path] = p
                    p.wait()
                    time.sleep(2)
            running[path] = True
            threading.Thread(target=runner, daemon=True).start()
            bot.reply_to(msg, f"🚀 {file_name} دەستپێکر.")
    except: pass

# --- STARTUP ---
